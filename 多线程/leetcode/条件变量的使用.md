[1195. 交替打印字符串](https://leetcode.cn/problems/fizz-buzz-multithreaded/description/)

卡住的点：关于当前是否已经达到n
解决方法：在唤醒阻塞线程时，加上是否达到n的判断，也就是该线程既要判断继续执行的条件，也要判断是否该退出的条件(让自己不要再阻塞了)

```cpp
class FizzBuzz {
private:
    int n;
    int now = 1;
    std::mutex mtx;
    std::condition_variable cv;
public:
    FizzBuzz(int n) {
        this->n = n;
    }

    // printFizz() outputs "fizz".
    void fizz(function<void()> printFizz) {
        while(now <= n){
            std::unique_lock<std::mutex> lock(mtx);
            cv.wait(lock,[this](){
                return this->now % 3 == 0 && this->now % 5 != 0 || this->now > n;
            });
            if(now <= n) printFizz();
            this->now++;
            cv.notify_all();
        }
    }

    // printBuzz() outputs "buzz".
    void buzz(function<void()> printBuzz) {
        while(now <= n){
            std::unique_lock<std::mutex> lock(mtx);
            cv.wait(lock,[this](){
                return this->now % 5 == 0 && this->now % 3 != 0 || this->now > n;
            });
            if(now <= n)printBuzz();
            this->now++;
            cv.notify_all();
        }
    }

    // printFizzBuzz() outputs "fizzbuzz".
	void fizzbuzz(function<void()> printFizzBuzz) {
        while(now <= n){
            std::unique_lock<std::mutex> lock(mtx);
            cv.wait(lock,[this](){
                return this->now % 3 == 0 && this->now % 5 == 0 || this->now > n;
            });
            if(now <= n)printFizzBuzz();
            this->now++;
            cv.notify_all();
        }
    }

    // printNumber(x) outputs "x", where x is an integer.
    void number(function<void(int)> printNumber) {
        while(now <= n){
            std::unique_lock<std::mutex> lock(mtx);
            cv.wait(lock,[this](){
                return this->now % 3 != 0 && this->now % 5 != 0 || this->now > n;
            });
            if(now <= n)printNumber(now);
            this->now++;
            cv.notify_all();
        }
    }
};
```
