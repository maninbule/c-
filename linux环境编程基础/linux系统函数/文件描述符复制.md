# dup函数
```c
#include <unistd.h>
int dup(int oldfd);
    作用：复制一个新的文件描述符
    fd=3, int fd1 = dup(fd),
    fd指向的是a.txt, fd1也是指向a.txt
    从空闲的文件描述符表中找一个最小的，作为新的拷贝的文件描述符
```
```c
#include <unistd.h>
#include <stdio.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <string.h>

int main(){
    int fd = open("a.txt",O_RDWR|O_CREAT,0664);
    if(fd == -1){
        perror("open: ");
        return -1;
    }
    int newfd = dup(fd); //newfd 和 fd具有相同的offset和mode，但是并不共享
    char *s1 = "hello,";
    int cnt = write(fd,s1,strlen(s1));
    if(cnt == -1){
        perror("write: ");
        return -1;
    }
    close(fd);
    char *s2 = "world\n";
    cnt = write(newfd,s2,strlen(s2));
    if(cnt == -1){
        perror("write: ");
        return -1;
    }
    close(newfd);



    return 0;
}
```
# dup2函数
```c
#include <unistd.h>
int dup2(int oldfd, int newfd);
    作用：重定向文件描述符
    oldfd 指向 a.txt, newfd 指向 b.txt
    调用函数成功后：newfd 和 b.txt 做close, newfd 指向了 a.txt
    也就是被fd进行指定描述符newfd进行复制，系统会把newfd做close，再进行重用
    oldfd 必须是一个有效的文件描述符
    oldfd和newfd值相同，相当于什么都没有做
```
```c
#include <unistd.h>
#include <stdio.h>
#include <sys/stat.h>
#include <sys/types.h>
#include <fcntl.h>
#include <string.h>

int main(){
    int fd1 = open("1.txt",O_RDWR|O_CREAT,0664);
    if(fd1 == -1){
        perror("open:");
        return -1;
    }
    int fd2 = open("2.txt",O_RDWR|O_CREAT,0664);
    if(fd2 == -1){
        perror("open: ");
        return -1;
    }
    int fd3 = dup2(fd1,fd2);
    if(fd3 == -1){
        perror("dup2: ");
        return -1;
    }
    char *s1 = "hello,";
    int cnt = write(fd1,s1,strlen(s1));
    if(cnt == -1){
        perror("write: ");
        return -1;
    }
    char *s2 = "world\n";
    cnt = write(fd2,s2,strlen(s2));
    if(cnt == -1){
        perror("write: ");
        return -1;
    }

    close(fd1);
    close(fd2);

    return 0;
}
```
